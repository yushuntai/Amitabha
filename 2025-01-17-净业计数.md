class NianFoCounter:
    def __init__(self):
        self.count = 0

    def recite(self, times=1):
        self.count += times
        print(f"累计念佛次数: {self.count}")

    def reset(self):
        self.count = 0
        print("计数已重置。")

if __name__ == "__main__":
    counter = NianFoCounter()

    while True:
        print("\n请输入对应的选项:")
        print("1: 念佛一次")
        print("2: 累计念佛指定次数")
        print("3: 重置计数")
        print("4: 退出")

        choice = input("选择: ").strip()

        if choice == "1":
            counter.recite()
        elif choice == "2":
            times = int(input("请输入念佛次数: ").strip())
            counter.recite(times)
        elif choice == "3":
            counter.reset()
        elif choice == "4":
            print("感谢使用，再见！")
            break
        else:
            print("无效的选择，请重试。")
